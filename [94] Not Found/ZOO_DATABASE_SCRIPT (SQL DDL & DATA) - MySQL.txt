CREATE TABLE HABITAT (
    HabitatID        VARCHAR(10) PRIMARY KEY,
    HabitatType      VARCHAR(50) NOT NULL,
    HabitatSize      INT NOT NULL,
    Climate          VARCHAR(20) NOT NULL CHECK (Climate IN ('Dry', 'Humid', 'Cold', 'Hot', 'Temperate')),
    MaximumCapacity  TINYINT NOT NULL CHECK (MaximumCapacity > 0 AND MaximumCapacity <= 30),
    Status           VARCHAR(50) NOT NULL CHECK (Status IN ('Operational', 'Under Renovation', 'Designated for Future Use', 'Under Maintenance'))
);

CREATE TABLE ANIMAL (
    AnimalID            VARCHAR(10) PRIMARY KEY,
    Species             VARCHAR(50) NOT NULL,
    Name                VARCHAR(50) NOT NULL,
    Gender              VARCHAR(10) NOT NULL CHECK (Gender IN ('Male', 'Female')),
    BirthDate           DATE NOT NULL,
    CurrentHealthStatus VARCHAR(50) NOT NULL,
    Diet                VARCHAR(20) NOT NULL CHECK (Diet IN ('Carnivore', 'Herbivore', 'Omnivore')),
    ArrivalDate         DATE NOT NULL,
    HabitatID           VARCHAR(10) NOT NULL,
    FOREIGN KEY (HabitatID) REFERENCES HABITAT(HabitatID)
);

CREATE TABLE ZOOKEEPER (
    ZookeeperID           VARCHAR(10) PRIMARY KEY,
    Name                  VARCHAR(100) NOT NULL,
    ExperienceLevel       VARCHAR(20) NOT NULL CHECK (ExperienceLevel IN ('Beginner', 'Intermediate', 'Advanced', 'Expert')),
    SpecialisedSkill      VARCHAR(50) NOT NULL,
    HabitatID             VARCHAR(10) NOT NULL,
    SupervisorID          VARCHAR(10),
    SupervisionStartDate  DATE NOT NULL,
    SupervisionEndDate    DATE,
    SupervisionLevel      TINYINT NOT NULL,
    FOREIGN KEY (HabitatID) REFERENCES HABITAT(HabitatID),
    FOREIGN KEY (SupervisorID) REFERENCES ZOOKEEPER(ZookeeperID)
);
